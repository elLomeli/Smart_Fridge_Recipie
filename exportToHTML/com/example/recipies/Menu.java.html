    @Override
    protected void onStart() {
        super.onStart();

        updateDate();

    private void updateDate() {
        // Obtiene la fecha actual
        Date currentDate = new Date();

        // Formatea la fecha en el formato deseado (por ejemplo, "dd/MM/yyyy")
        SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy", Locale.getDefault());
        String formattedDate = sdf.format(currentDate);

        // Actualiza el TextView con la nueva fecha
        dateTextView.setText(formattedDate);
    }

    private void scheduleDailyUpdate() {
        handler.postDelayed(new Runnable() {
            @Override
            public void run() {
                updateDate();

                // Programa la pr√≥xima actualizaci√≥n diaria
                scheduleDailyUpdate();
            }
        }, getNextMidnightMillis());
    }

    private void updateHour() {
        // Obtiene la hora actual
        Date currentTime = new Date();

        // Formatea la hora en el formato deseado (por ejemplo, "HH:mm:ss")
        SimpleDateFormat sdf = new SimpleDateFormat("HH:mm:ss", Locale.getDefault());
        String formattedHour = sdf.format(currentTime);

        // Actualiza el TextView con la nueva hora
        horaTextView.setText(formattedHour);
    }

    private void scheduleHourUpdate() {
        handler.postDelayed(new Runnable() {
            @Override
            public void run() {
                updateHour();

                // Programa la pr√≥xima actualizaci√≥n (cada segundo)
                scheduleHourUpdate();
            }
        }, 1000); // Actualizar cada segundo (1000 ms)
    }

    private long getNextMidnightMillis() {
        long currentTimeMillis = System.currentTimeMillis();
        long nextMidnightMillis = ((currentTimeMillis / 86400000) + 1) * 86400000; // Pr√≥xima medianoche en milisegundos
        return nextMidnightMillis - currentTimeMillis;
    }

    @Override
    protected void onActivityResult(int requestCode, int resultCode, Intent data) {
        super.onActivityResult(requestCode, resultCode, data);

        if (requestCode == REQUEST_CONNECT_DEVICE && resultCode == RESULT_OK) {
            // Obtiene la direcci√≥n MAC del dispositivo seleccionado
            String deviceAddress = data.getStringExtra("DEVICE_ADDRESS");

            // Conecta al dispositivo Bluetooth
            connectToBluetoothDevice(deviceAddress);
        } else if (requestCode == REQUEST_ENABLE_BT) {
            // Verifica si se han habilitado los permisos de Bluetooth
            if (resultCode == RESULT_OK) {
                // Los permisos de Bluetooth se han habilitado con √©xito, inicia la b√∫squeda de dispositivos
                Intent intent = new Intent(Menu.this, Dispositivos_Vinculados.class);
                startActivityForResult(intent, REQUEST_CONNECT_DEVICE);
            } else {
                // Los permisos de Bluetooth no se habilitaron, muestra un mensaje o realiza una acci√≥n adecuada
                Toast.makeText(this, "No se habilitaron los permisos de Bluetooth", Toast.LENGTH_SHORT).show();
            }
        }
    }

    private void connectToBluetoothDevice(String deviceAddress) {
        BluetoothDevice device = bluetoothAdapter.getRemoteDevice(deviceAddress);
        SharedPreferences.Editor editor = getSharedPreferences(PREFS_NAME, MODE_PRIVATE).edit();
        editor.apply();
        try {
            btSocket = device.createRfcommSocketToServiceRecord(MY_UUID);
            btSocket.connect();
            inStream = btSocket.getInputStream();

            // Iniciar un hilo para leer datos del flujo de entrada
            startBluetoothConnection();
            Toast.makeText(this, "Conectado Exitosamente", Toast.LENGTH_SHORT).show();

        } catch (IOException e) {
            // Manejar errores de conexi√≥n
            bluetoothbutton.setImageResource(R.drawable.bluetooth_off);
            Toast.makeText(this, "Desconectado Exitosamente", Toast.LENGTH_SHORT).show();
        }
    }

    private void startBluetoothConnection() {
        @SuppressLint("HandlerLeak") final Handler handler = new Handler() {
            public void handleMessage(Message msg) {
                if (msg.what == 0) {
                    String readMessage = (String) msg.obj;
                    // Dividir la cadena en datos de temperatura y humedad
                    String[] dataParts = readMessage.split(",");

                    if (dataParts.length >= 2) {
                        // La primera parte es la temperatura y la segunda parte es la humedad
                        temperaturaStr = dataParts[0]; // Supongamos que obtuviste la temperatura como una cadena
                        humedad = dataParts[1];
                        if (!temperaturaStr.isEmpty()) {
                            try {
                                Integer temperatura = Integer.parseInt(temperaturaStr); // Convierte la cadena a un n√∫mero
                                // Mostrar los datos en los TextView
                                textViewBluetoothData.setText(temperatura + " ¬∞C");
                                if (temperatura != temperaturaAnterior) {
                                    // La temperatura ha cambiado, actualiza la animaci√≥n
                                    if (temperatura < 1) {
                                        animationTemperaturta.setAnimation(R.raw.congelando);
                                        linearLayoutTemp.setBackgroundColor(ContextCompat.getColor(Menu.this, R.color.Congelando));

                                    } else if (temperatura >= 1 && temperatura <= 6) {
                                        animationTemperaturta.setAnimation(R.raw.termometrofrio);
                                        linearLayoutTemp.setBackgroundColor(ContextCompat.getColor(Menu.this, R.color.Perfecto));
                                        correoEnviado = true;
                                    } else if (temperatura >= 7 && temperatura <= 10) {
                                        linearLayoutTemp.setBackgroundColor(ContextCompat.getColor(Menu.this, R.color.Aceptable));
                                        animationTemperaturta.setAnimation(R.raw.termometrocaliente);
                                        correoEnviado = true;
                                    } else {
                                        animationTemperaturta.setAnimation(R.raw.termometrocaliente);
                                        linearLayoutTemp.setBackgroundColor(ContextCompat.getColor(Menu.this, R.color.Caliente));
                                    }
                                    // Inicia la animaci√≥n
                                    animationTemperaturta.setRepeatCount(LottieDrawable.INFINITE);
                                    animationTemperaturta.playAnimation();

                                    // Actualiza la temperatura anterior
                                    temperaturaAnterior = temperatura;

                                    final Handler handler = new Handler();
                                    handler.postDelayed(new Runnable() {
                                        @Override
                                        public void run() {
                                            // Verifica la temperatura aqu√≠
                                            if (temperaturaAnterior == temperatura) {
                                                tiempoActual += INTERVALO_TEMPORIZADOR / 1000; // Intervalo a segundos
                                            } else {
                                                tiempoActual = 0; // Reinicia el tiempo si la temperatura es menor de 10 grados
                                            }

                                            // Si el tiempo supera el l√≠mite y el correo no se ha enviado, env√≠a el correo
                                            if (tiempoActual >= TIEMPO_LIMITE && !correoEnviado) {
                                                enviarCorreo();
                                                correoEnviado = true; // Marca que el correo ha sido enviado
                                            }

                                            // Programa la siguiente verificaci√≥n
                                            handler.postDelayed(this, INTERVALO_TEMPORIZADOR);
                                        }
                                    }, INTERVALO_TEMPORIZADOR);
                                }


                            } catch (NumberFormatException e) {
                                // Manejar el caso en el que no se pudo convertir la cadena a un n√∫mero
                                // Puedes mostrar un mensaje de error, establecer una temperatura predeterminada, etc.
                            }
                        }
                    }
                }
            }
        };

        // Crear un hilo para leer datos del flujo de entrada
        Thread workerThread = new Thread(new Runnable() {
            public void run() {
                while (!Thread.currentThread().isInterrupted()) {
                    try {
                        int bytesAvailable = inStream.available();
                        if (bytesAvailable > 0) {
                            byte[] packetBytes = new byte[bytesAvailable];
                            inStream.read(packetBytes);
                            String readMessage = new String(packetBytes, 0, bytesAvailable);
                            bluetoothbutton.setImageResource(R.drawable.bluetooth_on);
                            // Env√≠a los datos al Handler para su procesamientoz
                            handler.obtainMessage(0, readMessage).sendToTarget();
                        }
                    } catch (IOException ex) {
                        bluetoothbutton.setImageResource(R.drawable.bluetooth_off);
                        break;
                    }
                }
            }
        });
        workerThread.start();
    }

    private void enviarCorreo() {
        String mSubject = "ALERTA DE TEMPERATURA";
        String mMessage = "PUEDE QUE TU REFIGERADOR ESTE ABIERTO, SI NO ES EL CASO, ES POSIBLE QUE ESTE FALLANDO YA QUE LA TEMPERATURA NO ESTA EN LOS RANGOS ACEPTABLES , LLAMA A UN TECNICO PARA QUE LO REVISE";
        JavaMailAPI javaMailAPI = new JavaMailAPI(this, email, mSubject, mMessage);

        javaMailAPI.execute();
    }
